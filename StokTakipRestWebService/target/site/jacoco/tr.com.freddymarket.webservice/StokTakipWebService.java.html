package tr.com.freddymarket.webservice;

import org.springframework.web.bind.annotation.*;
import java.util.*;

@RestController
@RequestMapping("/urun")
public class StokTakipWebService {
	
	record Urun (String ad, int stok) {}
	
	private static final List<Urun> URUN_LISTESI = new ArrayList<>();
	
	static { //Örnek ürünler
		URUN_LISTESI.add(new Urun("Bakliyat", 5));
		URUN_LISTESI.add(new Urun("Sebze", 23));
		URUN_LISTESI.add(new Urun("Icecek", 251));
	}
	
	@GetMapping("/")
	public List<Urun> listele(){ //Ürün listeleme metodu
		return URUN_LISTESI;
	}
	
	@PostMapping("/")
	public Urun ekle(@RequestBody Urun urun) { //Ürün ekleme metodu
		URUN_LISTESI.add(urun);
		return urun;
	}
	
	@DeleteMapping("/")
	public Urun sil(@RequestBody Urun urun) { /*Ürün silme metodu*/
		URUN_LISTESI.remove(urun);
		return urun;
		}
	
}
